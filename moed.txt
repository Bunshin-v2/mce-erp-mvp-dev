```typescript
async function getFinancialKPIs(orgId: ObjectId, startDate: Date, endDate: Date) {
  const pipeline = [
    // Total Project Value
    {
      $facet: {
        projectValue: [
          {
            $match: {
              clientId: orgId,
              status: { $in: ['ACTIVE_DESIGN', 'UNDER_CONSTRUCTION', 'COMMISSIONING'] },
              deletedAt: null
            }
          },
          {
            $group: {
              _id: null,
              totalValue: { $sum: '$budget.total' },
              count: { $sum: 1 }
            }
          }
        ],
        
        // Monthly Revenue
        monthlyRevenue: [
          {
            $match: {
              status: 'PAID',
              paidAt: {
                $gte: new Date(new Date().getFullYear(), new Date().getMonth(), 1),
                $lt: new Date(new Date().getFullYear(), new Date().getMonth() + 1, 1)
              },
              deletedAt: null
            }
          },
          {
            $group: {
              _id: null,
              revenue: { $sum: '$total' }
            }
          }
        ],
        
        // Outstanding Invoices
        outstandingInvoices: [
          {
            $match: {
              status: { $in: ['SENT', 'OVERDUE', 'PARTIALLY_PAID'] },
              deletedAt: null
            }
          },
          {
            $group: {
              _id: null,
              outstanding: { $sum: '$amountDue' },
              count: { $sum: 1 }
            }
          }
        ]
      }
    }
  ];
  
  return await Invoice.aggregate(pipeline);
}
```

```typescript
              _id: null,

        qualityScore: [

            $match: {

          {
            $group: {
              _id: null,
```

## 2. Real-Time Project Pipeline

```typescript
async function getProjectPipeline(orgId: ObjectId) {
  return await Project.aggregate([
    {
      $match: {
        orgId,
        deletedAt: null
      }
    },
    {
      $group: {
        _id: '$status',
        count: { $sum: 1 },
        totalValue: { $sum: '$budget.total' },
        avgProgress: { $avg: '$progress' }
      }
    },
    {
      $lookup: {
        from: 'milestones',
        let: { projectStatus: '$_id' },
        pipeline: [
          {
            $match: {
              $expr: {
                $and: [
                  { $eq: ['$$projectStatus', '$projectId'] },
                  {
                    $gte: [
                      '$dueAt',
                      new Date()
                    ]
                  },
                  {
                    $lte: [
                      '$dueAt',
                      new Date(Date.now() + 30 * 24 * 60 * 60 * 1000)
                    ]
                  }
                ]
              }
            }
          },
          { $count: 'count' }
        ],
        as: 'upcomingMilestones'
      }
    },
    {
      $project: {
        phase: '$_id',
        count: 1,
        value: '$totalValue',
        progress: { $round: ['$avgProgress', 0] },
        next30Days: {
          $ifNull: [{ $arrayElemAt: ['$upcomingMilestones.count', 0] }, 0]
        }
      }
    },
    {
      $sort: { count: -1 }
    }
  ]);
}
```

```typescript
async function getLiabilityAlerts() {
  const now = new Date();
  const thirtyDays = new Date(now.getTime() + 30 * 24 * 60 * 60 * 1000);
  const sixtyDays = new Date(now.getTime() + 60 * 24 * 60 * 60 * 1000);
  const ninetyDays = new Date(now.getTime() + 90 * 24 * 60 * 60 * 1000);
  
  return await Contract.aggregate([
    {
      $match: {
        status: 'ACTIVE',
        deletedAt: null,
        $or: [
          { endAt: { $lte: ninetyDays } },
          { renewalAt: { $lte: ninetyDays } },
          { bondExpiry: { $lte: ninetyDays } },
          { guaranteeExpiry: { $lte: ninetyDays } }
        ]
      }
    },
    {
      $project: {
        contractNumber: 1,
        title: 1,
        type: 1,
        counterpartyName: 1,
        expiryDate: {
          $min: ['$endAt', '$renewalAt', '$bondExpiry', '$guaranteeExpiry']
        },
        alertType: {
          $cond: {
            if: { $lte: ['$bondExpiry', ninetyDays] },
            then: 'Bond Maturity',
            else: {
              $cond: {
                if: { $lte: ['$endAt', ninetyDays] },
                then: 'Contract Expiry',
                else: 'Renewal Due'
              }
            }
          }
        },
        daysRemaining: {
          $divide: [
            { $subtract: ['$endAt', now] },
            1000 * 60 * 60 * 24
          ]
        }
      }
    },
    {
      $addFields: {
        severity: {
          $switch: {
            branches: [
              { case: { $lte: ['$daysRemaining', 14] }, then: 'HIGH' },
              { case: { $lte: ['$daysRemaining', 30] }, then: 'MEDIUM' },
              { case: { $lte: ['$daysRemaining', 60] }, then: 'LOW' }
            ],
            default: 'LOW'
          }
        }
      }
    },
    {
      $sort: { daysRemaining: 1 }
    }
  ]);
}
```

```typescript
async function getCashFlowProjection(months: number = 6) {
  const today = new Date();
  const futureDate = new Date(today.getTime() + months * 30 * 24 * 60 * 60 * 1000);
  
  return await Invoice.aggregate([
    {
      $match: {
        status: { $in: ['SENT', 'OVERDUE', 'PARTIALLY_PAID'] },
        dueAt: { $gte: today, $lte: futureDate },
        deletedAt: null
      }
    },
    {
      $group: {
        _id: {
          year: { $year: '$dueAt' },
          month: { $month: '$dueAt' }
        },
        expectedReceivables: { $sum: '$amountDue' },
        invoiceCount: { $sum: 1 }
      }
    },
    {
      $sort: { '_id.year': 1, '_id.month': 1 }
    },
    {
      $project: {
        period: {
          $concat: [
            { $toString: '$_id.year' },
            '-',
            {
              $cond: {
                if: { $lt: ['$_id.month', 10] },
                then: { $concat: ['0', { $toString: '$_id.month' }] },
                else: { $toString: '$_id.month' }
              }
            }
          ]
        },
        expectedReceivables: { $round: ['$expectedReceivables', 2] },
        invoiceCount: 1
      }
    }
  ]);
}
```

```typescript
async function getAgingAnalysis() {
  const today = new Date();
  
  return await Invoice.aggregate([
    {
      $match: {
        status: { $in: ['SENT', 'OVERDUE', 'PARTIALLY_PAID'] },
        deletedAt: null
      }
    },
    {
      $addFields: {
        daysOverdue: {
          $ceil: {
            $divide: [
              { $subtract: [today, '$dueAt'] },
              1000 * 60 * 60 * 24
            ]
          }
        }
      }
    },
    {
      $bucket: {
        groupBy: '$daysOverdue',
        boundaries: [-Infinity, 0, 30, 60, 90, Infinity],
        default: 'Unknown',
        output: {
          count: { $sum: 1 },
          totalAmount: { $sum: '$amountDue' },
          invoices: {
            $push: {
              code: '$code',
              client: '$clientId',
              amount: '$amountDue',
              daysOverdue: '$daysOverdue'
            }
          }
        }
      }
    },
    {
      $project: {
        category: {
          $switch: {
            branches: [
              { case: { $eq: ['$_id', -Infinity] }, then: 'Current' },
              { case: { $eq: ['$_id', 0] }, then: 'Current' },
              { case: { $eq: ['$_id', 30] }, then: '1-30 Days' },
              { case: { $eq: ['$_id', 60] }, then: '31-60 Days' },
              { case: { $eq: ['$_id', 90] }, then: '61-90 Days' },
              { case: { $eq: ['$_id', Infinity] }, then: '90+ Days' }
            ],
            default: 'Unknown'
          }
        },
        count: 1,
        totalAmount: { $round: ['$totalAmount', 2] }
      }
    }
  ]);
}
```

```typescript
async function getProjectRiskAssessment() {
  return await Project.aggregate([
    {
      $match: {
        status: { $in: ['ACTIVE_DESIGN', 'UNDER_CONSTRUCTION'] },
        deletedAt: null
      }
    },
    {
      $lookup: {
        from: 'tasks',
        let: { projectId: '$_id' },
        pipeline: [
          {
            $match: {
              $expr: { $eq: ['$projectId', '$$projectId'] },
              status: 'BLOCKED',
              deletedAt: null
            }
          },
          { $count: 'count' }
        ],
        as: 'blockedTasks'
      }
    },
    {
      $addFields: {
        blockedTaskCount: {
          $ifNull: [{ $arrayElemAt: ['$blockedTasks.count', 0] }, 0]
        },
        budgetVariance: {
          $subtract: [
            { $divide: ['$budget.spent', '$budget.total'] },
            { $divide: ['$progress', 100] }
          ]
        },
        isOverBudget: {
          $gt: ['$budget.spent', '$budget.total']
        },
        daysOverdue: {
          $cond: {
            if: { $gt: [new Date(), '$endAt'] },
            then: {
              $ceil: {
                $divide: [
                  { $subtract: [new Date(), '$endAt'] },
                  1000 * 60 * 60 * 24
                ]
              }
            },
            else: 0
          }
        }
      }
    },
    {
      $addFields: {
        riskScore: {
          $add: [
            { $multiply: ['$blockedTaskCount', 2] },
            { $cond: ['$isOverBudget', 5, 0] },
            { $multiply: ['$daysOverdue', 0.5] },
            {
              $switch: {
                branches: [
                  { case: { $eq: ['$riskLevel', 'CRITICAL'] }, then: 10 },
                  { case: { $eq: ['$riskLevel', 'HIGH'] }, then: 7 },
                  { case: { $eq: ['$riskLevel', 'MEDIUM'] }, then: 4 }
                ],
                default: 0
              }
            }
          ]
        }
      }
    },
    {
      $project: {
        code: 1,
        name: 1,
        status: 1,
        health: 1,
        riskLevel: 1,
        riskScore: { $round: ['$riskScore', 1] },
        blockedTasks: '$blockedTaskCount',
        isOverBudget: 1,
        daysOverdue: 1,
        budgetVariance: { $round: [{ $multiply: ['$budgetVariance', 100] }, 1] }
      }
    },
    {
      $sort: { riskScore: -1 }
    },
    {
      $limit: 20
    }
  ]);
}
```

```typescript
          { bondRequired: true },
          { guaranteeRequired: true },
          { type: 'INSURANCE' }
        ],
        deletedAt: null
      }
    },
    {
      $project: {
        liabilities: {
          $filter: {
            input: [
              {
                $cond: {
                  if: '$bondRequired',
                  then: {
                    category: 'FINANCIAL',
                    type: 'Performance Bond',
                    reference: '$contractNumber',
                    amount: '$bondAmount',
                    currency: '$currency',
                    expiryDate: '$bondExpiry',
                    riskLevel: 'MEDIUM'
                  },
                  else: '$$REMOVE'
                }
              },
              {
                $cond: {
                  if: '$guaranteeRequired',
                  then: {
                    category: 'FINANCIAL',
                    type: 'Bank Guarantee',
                    reference: '$contractNumber',
                    amount: '$guaranteeAmount',
                    currency: '$currency',
                    expiryDate: '$guaranteeExpiry',
                    riskLevel: 'MEDIUM'
                  },
                  else: '$$REMOVE'
                }
              },
              {
                $cond: {
                  if: { $eq: ['$type', 'INSURANCE'] },
                  then: {
                    category: 'FINANCIAL',
                    type: 'Insurance Policy',
                    reference: '$contractNumber',
                    counterparty: '$counterpartyName',
                    value: '$value',
                    currency: '$currency',
                    expiryDate: '$endAt',
                    riskLevel: 'HIGH'
                  },
                  else: '$$REMOVE'
                }
              }
            ],
            as: 'item',
            cond: { $ne: ['$$item', '$$REMOVE'] }
          }
        }
      }
    },
    { $unwind: '$liabilities' },
    { $replaceRoot: { newRoot: '$liabilities' } }
  ]);
}
```

## Liability Expiry Alert Schedule

```typescript
async function scheduleExpiryAlerts(daysBeforeExpiry: number[] = [30, 60, 90]) {
  const contracts = await Contract.find({
    status: 'ACTIVE',
    deletedAt: null
  });
  
  for (const contract of contracts) {
    for (const days of daysBeforeExpiry) {
      const alertDate = new Date(contract.endAt);
      alertDate.setDate(alertDate.getDate() - days);
      
      if (alertDate > new Date()) {
        await Notification.create({
          userId: contract.responsibleUserId,
          kind: 'contract_expiring',
          priority: days <= 30 ? 'URGENT' : days <= 60 ? 'HIGH' : 'MEDIUM',
          title: `Contract Expiring in ${days} Days`,
          message: `${contract.title} (${contract.contractNumber}) expires on ${contract.endAt.toLocaleDateString()}`,
          payload: {
            contractId: contract._id,
            contractNumber: contract.contractNumber,
            expiryDate: contract.endAt,
            daysRemaining: days
          },
          channels: ['EMAIL', 'WHATSAPP', 'IN_APP'],
          scheduledFor: alertDate,
          relatedEntityType: 'CONTRACT',
          relatedEntityId: contract._id
        });
      }
    }
  }
}
```

## Performance Optimization Tips

1. **Indexes**: Ensure all collections have proper indexes on frequently queried fields
2. **Caching**: Cache dashboard results for 5-15 minutes using Redis
3. **Pagination**: Always limit results for large datasets
4. **Aggregation**: Use `$match` early in pipeline to reduce document count
5. **Projection**: Only select needed fields in final `$project` stage
6. **Batch Processing**: Run liability alerts as scheduled jobs, not on-demand
