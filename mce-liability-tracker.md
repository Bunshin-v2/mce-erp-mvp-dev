```python
def escalate_alert(item, days_overdue):
    """
    Escalation ladder based on overdue duration
    """
    stakeholders = item.stakeholders.split(' | ')
    
    if days_overdue <= 7:
        # L1: Primary stakeholders only
        recipients = stakeholders[:2]  # e.g., Project Manager + Finance Director
    
    elif days_overdue <= 14:
        # L2: Add department heads
        recipients = stakeholders[:4]
    
    elif days_overdue <= 30:
        # L3: Add Vice Chairman
        recipients = stakeholders[:5] + ['Vice Chairman']
    
    else:  # > 30 days overdue
        # L4: Executive escalation
        recipients = ['Chairman', 'Vice Chairman'] + stakeholders
        priority = 'CRITICAL'
        channels = ['email', 'sms', 'whatsapp']  # Multi-channel
    
    send_notification(item, recipients, priority, channels)
```
